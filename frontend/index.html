  <input id="inp" style="width:80%" placeholder="Вопрос..." />

  <button onclick="send()">Send</button>

<script>

const chat = document.getElementById("chat");

const uid = localStorage.USER_ID ||= crypto.randomUUID();

let ws, buf="";


function connect() {

  ws = new WebSocket(`ws://${location.host}/ws`);

  ws.onmessage = e => {

    const m = JSON.parse(e.data);

    if(m.type==="token") buf+=m.text;

    if(m.type==="done") buf+="\n\n";

    if(m.type==="cleared") buf="";

    chat.textContent = buf;

  };

}

connect();


function send(){

  ws.send(JSON.stringify({type:"chat", user_id:uid, text:inp.value}));

  inp.value="";

}

function clearChat(){

  ws.send(JSON.stringify({type:"clear", user_id:uid}));

}

</script>
